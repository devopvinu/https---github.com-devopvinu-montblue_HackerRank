                        public class SockMerchant {
                        
                            // Function to count the number of pairs of socks with matching colors
                            static int sockMerchant(int n, int[] ar) {
                                // Map to store the count of each color
                                Map colorCount = new HashMap<>();
                        
                                // Count the occurrences of each color
                                for (int color : ar) {
                                    colorCount.put(color, colorCount.getOrDefault(color, 0) + 1);
                                }
                        
                                // Count the number of pairs for each color
                                int totalPairs = 0;
                                for (int count : colorCount.values()) {
                                    totalPairs += count / 2;
                                }
                        
                                return totalPairs;
                            }
                        
                            public static void main(String[] args) {
                                Scanner scanner = new Scanner(System.in);
                        
                                // Read the number of socks
                                int n = scanner.nextInt();
                        
                                // Read the colors of the socks
                                int[] ar = new int[n];
                                for (int i = 0; i < n; i++) {
                                    ar[i] = scanner.nextInt();
                                }
                        
                                // Call the sockMerchant function and print the result
                                int result = sockMerchant(n, ar);
                                System.out.println(result);
                        
                                scanner.close();
                            }
                        }

def breaking_records(scores):
    high_score = scores[0]
    low_score = scores[0]
    high_count = 0
    low_count = 0

    for i in range(1, len(scores)):
        if scores[i] > high_score:
            high_score = scores[i]
            high_count += 1
        elif scores[i] < low_score:
            low_score = scores[i]
            low_count += 1

    return [high_count, low_count]

def quickSort(arr):
    left = []
    equal = []
    right = []
    pivot = arr[0]
    
    for el in arr:
        if el < pivot:
            left.append(el)
        elif el == pivot:
            equal.append(el)

    return left + equal + right

def minimumNumber(n, password):
    res = 0

    if not any(x in numbers for x in password):
        res += 1

        res += 1

    if not any(x in upper_case for x in password):
        res += 1

    if not any(x in special_characters for x in password):

    if len(password) < 6:
